    <script>
        // Navigation
        function switchTab(tabId) {
            document.querySelectorAll('.view-section').forEach(el => el.classList.add('hidden'));
            document.getElementById(`view-${tabId}`).classList.remove('hidden');
            document.querySelectorAll('.nav-btn').forEach(btn => {
                btn.classList.remove('tab-active', 'text-muji-brown');
                btn.classList.add('text-gray-400');
                if(btn.dataset.target === tabId) { btn.classList.add('tab-active', 'text-muji-brown'); btn.classList.remove('text-gray-400'); }
            });
            if(tabId === 'wallet') renderExpenses();
        }

        // Wallet
        let expenses = JSON.parse(localStorage.getItem('tainan_expenses')) || [];
        function addExpense() {
            const name = document.getElementById('expense-name').value;
            const amountStr = document.getElementById('expense-amount').value;
            if(!name || !amountStr) return alert('請輸入內容');
            let amount = 0;
            try { amount = Function('"use strict";return (' + amountStr + ')')(); } catch { return alert('金額格式錯誤'); }
            expenses.unshift({ id: Date.now(), name, amount, date: new Date().toLocaleDateString() });
            localStorage.setItem('tainan_expenses', JSON.stringify(expenses));
            renderExpenses();
            document.getElementById('expense-name').value = '';
            document.getElementById('expense-amount').value = '';
        }
        function renderExpenses() {
            const list = document.getElementById('expense-list');
            let total = 0;
            list.innerHTML = expenses.map(e => {
                total += e.amount;
                return `<div class="bg-white p-3 rounded-lg shadow-sm flex justify-between items-center"><div><p class="font-bold">${e.name}</p><p class="text-xs text-gray-400">${e.date}</p></div><div class="flex items-center gap-3"><span class="font-mono font-bold text-muji-brown">NT$ ${e.amount}</span><button onclick="deleteExpense(${e.id})" class="text-gray-300"><i class="fa-solid fa-trash"></i></button></div></div>`;
            }).join('');
            document.getElementById('total-expense').innerText = `NT$ ${total}`;
        }
        function deleteExpense(id) {
            if(confirm('刪除?')) { expenses = expenses.filter(e => e.id !== id); localStorage.setItem('tainan_expenses', JSON.stringify(expenses)); renderExpenses(); }
        }
        function clearExpenses() { if(confirm('清空?')) { expenses = []; localStorage.setItem('tainan_expenses', JSON.stringify(expenses)); renderExpenses(); } }

        // Lists
        const checkItems = ['身分證 / 健保卡', '手機充電器 / 行動電源', '個人常備藥品', '雨具 / 折疊傘', '好穿的走路鞋', '泳衣/泳褲 (汗蒸幕用)'];
        function renderChecklist() {
            const saved = JSON.parse(localStorage.getItem('tainan_checklist')) || {};
            const container = document.getElementById('checklist-container');
            container.innerHTML = checkItems.map(item => `
                <label class="flex items-center space-x-3 p-2 hover:bg-gray-50 rounded cursor-pointer">
                    <input type="checkbox" class="w-5 h-5 accent-muji-brown rounded" onchange="toggleCheck('${item}')" ${saved[item] ? 'checked' : ''}>
                    <span>${item}</span>
                </label>
            `).join('');
        }
        function toggleCheck(item) {
            const saved = JSON.parse(localStorage.getItem('tainan_checklist')) || {};
            saved[item] = !saved[item];
            localStorage.setItem('tainan_checklist', JSON.stringify(saved));
        }
        function saveMemo() { localStorage.setItem('tainan_memo', document.getElementById('memo-area').value); }

        // Init
        window.onload = function() {
            switchTab('plan');
            renderChecklist();
            document.getElementById('memo-area').value = localStorage.getItem('tainan_memo') || '';
        }
    </script>
